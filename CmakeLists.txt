cmake_minimum_required(VERSION 3.30.4)

project(Puzzle)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt5 COMPONENTS
        Core
        Gui
        Widgets
        REQUIRED)

add_executable(Puzzle src/main.cpp)
target_link_libraries(Puzzle 
        Qt5::Core
        Qt5::Gui
        Qt5::Widgets)
